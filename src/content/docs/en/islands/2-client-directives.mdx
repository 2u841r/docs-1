1. STARTING CODE:

```astro title="src/pages/products/[id].astro"
---
import Layout from "../../layouts/Layout.astro";
import products from "../../data/products.json";

export function getStaticPaths() {
  const pages = products.map((product) => ({
    params: { id: product.id },
    props: { product },
  }));

  return pages;
}

const { product } = Astro.props;
---

<Layout title={product.title}>
  <article class="flex flex-col sm:flex-row sm:gap-8">
    <img
      class="w-full mb-4 sm:mb-0 rounded-md"
      src={product.image}
      alt={product.title}
    />
    <div class="flex flex-col gap-4 w-full">
      <h1 class="text-xl font-semibold sm:text-3xl">{product.title}</h1>
      <p>$ {product.price.toFixed(2)}</p>
      <p>{product.description}</p>
    </div>
  </article>
</Layout>
```

2. ADD THE `<AddToCart />` component CODE

```astro title="src/pages/products/[id].astro" ins={4, 28-35}
---
import Layout from "../../layouts/Layout.astro";
import products from "../../data/products.json";
import { AddToCart } from "../../components/AddToCart";

export function getStaticPaths() {
  const pages = products.map((product) => ({
    params: { id: product.id },
    props: { product },
  }));

  return pages;
}

const { product } = Astro.props;
---

<Layout title={product.title}>
  <article class="flex flex-col sm:flex-row sm:gap-8">
    <img
      class="w-full mb-4 sm:mb-0 rounded-md"
      src={product.image}
      alt={product.title}
    />
    <div class="flex flex-col gap-4 w-full">
      <h1 class="text-xl font-semibold sm:text-3xl">{product.title}</h1>
      <p>$ {product.price.toFixed(2)}</p>
      <AddToCart
        product={{
          id: product.id,
          name: product.title,
          price: product.price,

    </div>
  </article>
```

3. Hydrate the component

```astro title="src/pages/products/[id].astro" ins={29}
---
import Layout from "../../layouts/Layout.astro";
import products from "../../data/products.json";
import { AddToCart } from "../../components/AddToCart";

export function getStaticPaths() {
  const pages = products.map((product) => ({
    params: { id: product.id },
    props: { product },
  }));

  return pages;
}

const { product } = Astro.props;
---

<Layout title={product.title}>
  <article class="flex flex-col sm:flex-row sm:gap-8">
    <img
      class="w-full mb-4 sm:mb-0 rounded-md"
      src={product.image}
      alt={product.title}
    />
    <div class="flex flex-col gap-4 w-full">
      <h1 class="text-xl font-semibold sm:text-3xl">{product.title}</h1>
      <p>$ {product.price.toFixed(2)}</p>
      <AddToCart
       client:load
        product={{
          id: product.id,
          name: product.title,
          price: product.price,
          image: product.image,
        }}
      />
      <p>{product.description}</p>
    </div>
  </article>
</Layout>
```

### Analyze the Pattern

Given the following `.astro` script:

```astro title="src/pages/about.astro"
---
const operatingSystem = "Linux";
const quantity = 3;
const footwear = "boots";
const student = false;
---
```


For each Astro template expression, can you predict the HTML (if any!) that will be sent to the browser? Click to reveal if you're right!

1.  `<p>{operatingSystem}</p>`

    <p>
      <Spoiler>`<p>Linux</p>`</Spoiler>
    </p>

2.  `{student && <p>I am still in school.</p>}`

    <p>
      <Spoiler>Nothing will display because `student` evaluates to false.</Spoiler>
    </p>

3.  `<p>I have {quantity + 8} pairs of {footwear}</p>`

    <p>
      <Spoiler>`<p>I have 11 pairs of boots</p>`</Spoiler>
    </p>
